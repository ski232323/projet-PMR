Navigation par Temps

let tempsCumules = {}; let positionPrecedente = null; // Fonction pour démarrer le chronomètre document.getElementById("demarrer").addEventListener("click", function() { const select = document.getElementById("itineraire"); const idItineraire = select.value; if (!idItineraire) { alert("Veuillez sélectionner un itinéraire."); return; } itineraireSelectionne = itineraireData[idItineraire]; positionPrecedente = itineraireSelectionne[0].depart; calculerTempsCumules(); document.getElementById("demarrer").disabled = true; document.getElementById("arret").disabled = false; document.getElementById("position").textContent = `Position estimée : ${positionPrecedente}`; intervalId = setInterval(mettreAJourTimer, 1000); }); // Fonction pour arrêter le chronomètre document.getElementById("arret").addEventListener("click", function() { clearInterval(intervalId); document.getElementById("demarrer").disabled = false; document.getElementById("arret").disabled = true; }); // Fonction pour calculer les temps cumulés function calculerTempsCumules() { tempsCumules = {}; let tempsTotal = 0; itineraireSelectionne.forEach(segment => { tempsTotal += segment.temps; tempsCumules[segment.arrivee] = tempsTotal; }); } // Fonction pour mettre à jour le timer et la position function mettreAJourTimer() { tempsEcoule += 1; const minutes = Math.floor(tempsEcoule / 60); const secondes = tempsEcoule % 60; document.getElementById("timer").textContent = `Temps écoulé : ${minutes} min ${secondes} s`; // Estimer la position let positionEstimee = itineraireSelectionne[0].depart; for (const [point, tempsCumule] of Object.entries(tempsCumules)) { if (tempsEcoule / 60 <= tempsCumule) { positionEstimee = point; break; } } // Afficher la flèche si la position a changé if (positionEstimee !== positionPrecedente) { document.getElementById("position").innerHTML = ` Position estimée : ${positionPrecedente} → ${positionEstimee} `; positionPrecedente = positionEstimee; } }
